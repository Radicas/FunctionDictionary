```sql
-- 备份现有数据
CREATE TABLE functions_backup AS SELECT * FROM functions;

-- 创建新表
CREATE TABLE projects (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    path TEXT NOT NULL UNIQUE,
    description TEXT,
    language TEXT,
    framework TEXT,
    status TEXT DEFAULT 'active',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    update_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- 重构functions表
DROP TABLE functions;
CREATE TABLE functions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER,
    file_id INTEGER,
    function_name TEXT NOT NULL,
    full_name TEXT,
    signature TEXT,
    return_type TEXT,
    parameters TEXT,
    description TEXT,
    code_snippet TEXT,
    line_start INTEGER,
    line_end INTEGER,
    complexity INTEGER,
    tags TEXT,
    category TEXT,
    ai_analysis TEXT,
    custom_notes TEXT,
    status TEXT DEFAULT 'active',
    create_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    update_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (project_id) REFERENCES projects(id)
);

-- 迁移旧数据
INSERT INTO functions (function_name, description, create_time)
SELECT key, value, create_time FROM functions_backup;

-- 创建索引
CREATE INDEX idx_functions_project ON functions(project_id);
CREATE INDEX idx_functions_name ON functions(function_name);
```

```cpp
class IncrementalAnalyzer {
public:
    void analyzeProject(int projectId) {
        QVector<FileInfo> files = getProjectFiles(projectId);
        
        for (const FileInfo& file : files) {
            if (isFileChanged(file)) {
                analyzeFile(file.id);
            }
        }
    }
    
private:
    bool isFileChanged(const FileInfo& file) {
        QFileInfo fileInfo(file.filePath);
        QDateTime lastModified = fileInfo.lastModified();
        
        return lastModified > file.lastAnalyzedTime;
    }
};
```
